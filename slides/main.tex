% --- fonts: libertinus body + inconsolata mono -------------------------------
% fontspec is already pulled in by metropolis (lualatex toolchain). Overrides
% must come AFTER \usetheme{metropolis} so they win against Fira.
\usepackage{fontspec}
\setsansfont{Libertinus Sans}[Ligatures=TeX]
\setmainfont{Libertinus Serif}[Ligatures=TeX]
% Inconsolata's CTAN package ships its OTF as "Inconsolatazi4-*" (the "zi4"
% variant — zen + inconsolata + 4 weights — see CTAN: tex-archive/fonts/inconsolata).
% fontspec will not resolve the name "Inconsolata" against this filename; we
% have to spell out the family + weights explicitly.
\setmonofont{Inconsolatazi4}[
  Extension   = .otf,
  UprightFont = *-Regular,
  BoldFont    = *-Bold,
  Scale       = 0.95
]
