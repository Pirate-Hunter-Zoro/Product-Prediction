\section{Problem}

\begin{frame}{Task}
  \begin{block}{Goal}
    Given a shopping session, rank the next product the user will interact with.
  \end{block}

  \vspace{0.4em}
  \begin{columns}[T,onlytextwidth]
    \column{0.52\textwidth}
      \textbf{Input}
      \begin{itemize}
        \item Ordered product history: \(\left[i_1, i_2, \ldots, i_t\right]\)
        \item Session id used as the user field for RecBole
        \item Product metadata: title, brand, color, price
      \end{itemize}

    \column{0.43\textwidth}
      \textbf{Evaluation}
      \begin{itemize}
        \item Full-sort ranking over the item vocabulary
        \item \metric{MRR@100}, \metric{Recall@100}, \metric{NDCG@100}
        \item Reported overall and per locale
      \end{itemize}
  \end{columns}

  \vspace{0.6em}
  \footnotesize Dataset and task follow Amazon-M2 Task 1 from the KDD Cup 2023 multilingual recommendation challenge.
\end{frame}

\section{Novel Method}

\begin{frame}{Motivation}
  \begin{block}{Amazon-M2 paper, Table 5 (Task 2 ablation, ES/FR/IT supervised)}
    The authors initialize the item-embedding table with Multilingual Sentence-BERT \emph{title} vectors (variants SRGNNF, GRU4RecF). MRR@100 and Recall@100 drop on every locale tested. Their conclusion: \textit{``how to effectively utilize textual features remains an open question.''}

    \vspace{0.2em}
    \begin{center}
    \scriptsize
    \begin{tabular}{lcc@{\hskip 1.5em}lcc}
      \toprule
      Variant & \metric{MRR} & \metric{Recall} & Variant & \metric{MRR} & \metric{Recall} \\
      \midrule
      SRGNN              & 0.2640 & 0.5429 & GRU4Rec++         & 0.2564 & 0.5856 \\
      \quad + title (F)  & 0.2303 & 0.4976 & \quad + title (F) & 0.2427 & 0.5454 \\
      \bottomrule
    \end{tabular}
    \end{center}
  \end{block}

  \vspace{0.3em}
  \begin{alertblock}{Our angle --- un-naive feature fusion}
    \scriptsize
    \begin{tabular}{@{}p{0.26\linewidth}p{0.33\linewidth}p{0.33\linewidth}@{}}
      \toprule
                              & Paper (SRGNNF / GRU4RecF)        & This work (NovelModel) \\
      \midrule
      Item-id embedding init  & MiniLM title vectors             & Random (RecBole default) \\
      How attributes enter    & Baked into the id-embedding row  & Separate per-item bag via cross-attention \\
      Attribute set           & Title only                       & Title, brand, color, price bin \\
      \bottomrule
    \end{tabular}
  \end{alertblock}
\end{frame}

\begin{frame}{Attribute Construction}
  \begin{columns}[T,onlytextwidth]
    \column{0.5\textwidth}
      \textbf{Text attributes}
      \begin{itemize}
        \item Title, brand, and color
        \item Encoded with \code{paraphrase-multilingual-MiniLM-L12-v2}
        \item 384-dimensional, L2-normalized vectors
        \item Aligned to RecBole internal item ids
      \end{itemize}

    \column{0.45\textwidth}
      \textbf{Price}
      \begin{itemize}
        \item 32 train-only quantile bins
        \item Learned embedding per bin
        \item Robust to extreme catalog price outliers
        \item Reserved sentinel for padding or missing bins
      \end{itemize}
  \end{columns}

  \vspace{0.8em}
  \begin{block}{Catalog coverage}
    Every item referenced by the session data has a catalog row, so the model does not need an unknown-attribute fallback.
  \end{block}
\end{frame}

\begin{frame}[fragile]{NovelModel forward walkthrough}
\vspace{-0.6em}
\scriptsize
\begin{verbatim}
def forward(self, item_seq, item_seq_len):
    # item_seq: (B, T) item ids, 0 = PAD.  item_seq_len: (B,) real lengths.
    x          = self.emb_dropout(self.item_embedding(item_seq))   # (B, T, H)
    gru_out, _ = self.gru(x)                # (B, T, H); 2nd return (n_layers, B, H) discarded
    state      = self.gather_indexes(gru_out, item_seq_len - 1)    # (B, H)
    K = len(self.active_item_attributes)
    proj = []
    for attr in self.active_item_attributes:                       # iterates 1..4 active slots
        if   attr == `title': proj.append(self.title_proj(self.title_embeddings))
        elif attr == `brand': proj.append(self.brand_proj(self.brand_embeddings))
        elif attr == `color': proj.append(self.color_proj(self.color_embeddings))
        elif attr == `price': proj.append(self.price_embedding(self.price_bin_idx))
    attr_bag = torch.stack(proj, dim=1)                            # (n_items, K, H)
    sess_bag = attr_bag[item_seq].flatten(1, 2)                    # (B, T*K, H)
    pad_mask = (item_seq == 0).repeat_interleave(K, dim=1)         # (B, T*K)
    attn, _  = self.cross_attn(state.unsqueeze(1), sess_bag, sess_bag,
                               key_padding_mask=pad_mask)          # (B, 1, H)
    return state + attn.squeeze(1)                                 # (B, H)
\end{verbatim}
\scriptsize Logits in \code{calculate\_loss}: \code{output @ self.item\_embedding.weight.T}, shape \((B,\ \texttt{n\_items})\). The \code{item\_embedding} on line 3 above and the score node of the diagram are the same tensor.
\end{frame}

\section{Results}

\begin{frame}{Evaluation Metrics}
  Each test session has exactly one held-out next item. Let \(r_s\) be the rank of that item in the model's full-vocabulary ranking for session \(s\); a session contributes \(0\) when \(r_s > 100\). All three metrics are means over the \(N\) test sessions.

  \vspace{0.5em}
  \begin{itemize}
    \item \metric{MRR@100} \(=\ \tfrac{1}{N}\sum_{s} \tfrac{1}{r_s}\,\mathbf{1}[r_s \le 100]\). Reciprocal rank --- rank 1 scores 1.0, rank 10 scores 0.1, rank 100 scores 0.01. Punishes slipping down the list.

    \item \metric{Recall@100} \(=\ \tfrac{1}{N}\sum_{s} \mathbf{1}[r_s \le 100]\). Equivalent to HitRate@100 here (single positive per session): did the true item make the top 100, yes or no.

    \item \metric{NDCG@100} \(=\ \tfrac{1}{N}\sum_{s} \tfrac{1}{\log_2(r_s+1)}\,\mathbf{1}[r_s \le 100]\). Log-discounted hit; same shape as MRR but gentler --- rank 1 scores 1.0, rank 10 scores \(\approx 0.29\), rank 100 scores \(\approx 0.15\).
  \end{itemize}

  \vspace{0.4em}
  \footnotesize Recall is the easiest to satisfy; MRR is the strictest. NDCG sits between them.
\end{frame}

\section{Conclusion}

\begin{frame}{Conclusion}
  \begin{block}{Deliverables}
    \scriptsize
    \begin{itemize}\setlength\itemsep{0.15em}
      \item Reproducible RecBole pipeline: preprocessing $\to$ training $\to$ per-locale evaluation, seeded 80/10/10 split across all six locales.
      \item GRU4Rec, NARM, and a hand-rolled Popularity baseline; per-locale numbers reach paper-ballpark on the UK/DE/JP weighted row.
      \item NovelModel: custom \code{SequentialRecommender} subclass with cross-attention over a title / brand / color / price-bin attribute bag, plus a slot-toggle ablation surface wired through \code{train.py}.
    \end{itemize}
  \end{block}

  \begin{block}{Limitations}
    \scriptsize
    \begin{itemize}\setlength\itemsep{0.3em}
      \item Our six-locale ``Overall'' $\ne$ the paper's UK/DE/JP-only ``Overall'' --- different universes; both reported with scope labels.
      \item GRU4Rec sits ${\sim}10\text{--}12\%$ behind paper Table 3 GRU4Rec++ on the UK/DE/JP-weighted row --- seeded-split + augmentation gap, not a bug.
      \item NovelModel full-test evaluation was still finishing at slide-cut: validation MRR@100 ${\approx}\,0.181$ at epoch 20.
      \item Paper Popularity (MRR 0.2426) is unreproducible: their protocol's definition of ``most popular'' differs from ours (per-locale vocabulary, leave-one-out test, or the official KDD Cup test set), but the release does not spell out which.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Future Work}
  \begin{block}{Where this goes next}
    \begin{itemize}\setlength\itemsep{0.4em}
      \item \textbf{Recall-targeted variant.} Swap the GRU for a CORE or MGS encoder under the same attribute-bag fusion --- target their higher Recall@100 (${\sim}0.55$ vs GRU4Rec++'s 0.38).
      \item \textbf{Slot-toggle ablations.} Title only $\to$ +brand $\to$ +color $\to$ +price. Infrastructure is already wired through \code{train.py --slots} and \code{run\_training.sbatch}; only the cluster jobs remain. Direct comparison to paper Table 5's title-only failure case.
    \end{itemize}
  \end{block}

\end{frame}
